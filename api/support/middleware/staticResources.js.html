Build middleware for serving static resources.

This middleware uses koa-static.

method
-> Function

Option Type Description
options Object Configuration options.
options.folder String The filesystem path (relative to the Waigo app folder) from which to serve resources.
[options.maxage] Integer Browser cache max-age in milliseconds (default is `0`).
[options.hidden] Boolean Whether to serve hidden resources too (default is `false`).
[options.index] String The index file (default is `index.html`).
[options.defer] Boolean if `true` then will serve after `yield next`, allowing downstream middleware to respond first (default is `true`). 
module.exports = function(options) {
  var moreOptions = _.extend({
    maxage: 0,
    hidden: false,
    index: 'index.html',
    defer: false
  }, options.options);

  return require('koa-static')(path.join(waigo.getAppFolder(), options.folder), moreOptions);
};
